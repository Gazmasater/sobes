func MigratePersonSchema(ctx context.Context, db *gorm.DB) {

	err := db.AutoMigrate(&PersonMigration{})
	if err != nil {
		logger.Fatalf(ctx, "failed to migrate Person schema: %v", err)
	}

	// –°–æ–∑–¥–∞–Ω–∏–µ —Å–æ—Å—Ç–∞–≤–Ω–æ–≥–æ –∏–Ω–¥–µ–∫—Å–∞ –¥–ª—è –ø–æ–ª–µ–π Name –∏ Surname

	}
}
